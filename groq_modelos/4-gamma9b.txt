from groq import Groq

client = Groq()
completion = client.chat.completions.create(
    model="gemma2-9b-it",
    messages=[
      {
        "role": "user",
        "content": ""
      }
    ],
    temperature=1,
    max_completion_tokens=8192,
    top_p=1,
    stream=False,
    response_format={"type": "json_object"},
    stop=None
)

print(completion.choices[0].message)
